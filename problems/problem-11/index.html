def print_grid(grid):
    for row in grid:
        print(" ".join(row))


def multiply_list(s: list):
    # multiply_list(["11","7"]) -> 77
    product = 1
    for i in s:
        product = product * int(i)
    return product


string = """08 02 22 97 38 15 00 40 00 75 04 05 07 78 52 12 50 77 91 08
49 49 99 40 17 81 18 57 60 87 17 40 98 43 69 48 04 56 62 00
81 49 31 73 55 79 14 29 93 71 40 67 53 88 30 03 49 13 36 65

# convert into 2d-array
row_strings = string.split("\n")
grid = []
for row_string in row_strings:
    grid.append(row_string.split())


maxNum = 0
# Attempt to find the greatest products of two adjacent numbers horizontally
max_value = 0
n = 4
for k in range(len(grid)):
    s = grid[k]
    for i in range(len(s)):
        try:
            if multiply_list(s[i : i + n]) > max_value:
                max_value = multiply_list(s[i : i + n])
                # print(max_value, s[i : i + n])
        except:
            pass
        if max_value > maxNum:
            maxNum = max_value

# Attempt to find the greatest products of two adjacent numbers vertically
max_value = 0
n = 4
for k in range(len(grid)):
    s = grid[k]
    for i in range(len(s)):
        try:
            pair = [grid[x][i] for x in range(k, k + n)]
            if multiply_list(pair) > max_value:
                max_value = multiply_list(pair)
                # print(max_value, pair)
        except:
            pass
        if max_value > maxNum:
            maxNum = max_value

# Attempt to find the greatest products of two adjacent numbers up-right
max_value = 0
n = 4
for k in range(len(grid)):
    s = grid[k]
    for i in range(len(s)):
        # Move from Index(1,1) to Index(0,2)
        # (4,0)->(3,1)->(2,2)->(1,3)->(0,4)
        # (k-0,i+0)->(k-1,i+1)->...-> (k-(n-1),i+(n-1))
        try:
            pair = []
            for x in range(n):
                assert k - x >= 0
                pair.append(grid[k - x][i + x])
            if multiply_list(pair) > max_value:
                max_value = multiply_list(pair)
                # print(max_value, pair)
        except:
            pass
        if max_value > maxNum:
            maxNum = max_value
# Attempt to find the greatest product of two adjacent numbers down-right
max_value = 0
n = 4
for k in range(len(grid)):
    s = grid[k]
    for i in range(len(s)):
        # Move from Index(0,0) to Index(1,1)
        # (0,0) -> (1,1) -> (2,2) -> (3,3) -> (4,4)
        # (k,i) -> (k+1,i+1)
        try:
            pair = []
            for x in range(n):
                assert k + x < len(grid)
                pair.append(grid[k + x][i + x])
            if multiply_list(pair) > max_value:
                max_value = multiply_list(pair)
                # print(max_value, pair)
        except:
            pass
        if max_value > maxNum:
            maxNum = max_value

print(maxNum)
